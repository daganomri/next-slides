# Variant Animations

<Grid columns={2} style={{gap:20}}>

```tsx showLineNumbers highlightLines=3,13,15
const listVariants = {
  hidden: {...},
  show: { transition: { staggerChildren: 0.1 } },
};
const itemVariants = {
  hidden: { opacity: 0, y: 20 },
  show: { opacity: 1, y: 0 },
};

const items = ["React Spring", "React Transition Group", "React Motion"];

return (
  <motion.ul variants={listVariants} initial="hidden" animate="show">
    {items.map((item) => (
      <motion.li key={item} variants={itemVariants}>
        {item}
      </motion.li>
    ))}
  </motion.ul>
);
```

# Shared Layout Animations

<Grid columns={2} style={{gap:20}}>

```tsx showLineNumbers
// Wrapper
const [selected, setSelected] = React.useState(colors[0]);

<AnimateSharedLayout>
  <ul>
    {colors.map((color) => (
      <Item
        key={color}
        color={color}
        isSelected={selected === color}
        onClick={() => setSelected(color)}
      />))}
  </ul>
</AnimateSharedLayout>

// Item
<li onClick={onClick} style={{ backgroundColor: color }}>
  {isSelected && (
    <motion.div
      layoutId="outline"
      initial={false}
      animate={{ borderColor: color }}
      transition={spring}
    />)}
</li>
```
